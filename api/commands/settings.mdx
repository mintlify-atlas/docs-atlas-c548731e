Commands for managing app settings, file operations, and system utilities.

## get_setting

Get a setting value by key.

```typescript
const value = await invoke<string | null>('get_setting', { key: string })
```

<ParamField path="key" type="string" required>
  Setting key name
</ParamField>

<ResponseField name="value" type="string | null">
  Setting value or null if not found
</ResponseField>

### Usage Example

```typescript
const downloadPath = await invoke<string | null>('get_setting', {
  key: 'download_path'
})

if (downloadPath) {
  console.log(`Download path: ${downloadPath}`)
} else {
  console.log('Using default download path')
}
```

## get_all_settings

Get all settings as key-value pairs.

```typescript
const settings = await invoke<Setting[]>('get_all_settings')
```

<ResponseField name="Setting[]" type="array">
  <Expandable title="Setting properties">
    <ResponseField name="key" type="string">
      Setting key
    </ResponseField>
    
    <ResponseField name="value" type="string">
      Setting value
    </ResponseField>
  </Expandable>
</ResponseField>

### Usage Example

```typescript

  settings.map(s => [s.key, s.value])
```

## open_folder

Open a folder in the system file explorer, optionally highlighting a file.

```typescript
await invoke('open_folder', {
  path: string,
  file_name?: string
})
```

<ParamField path="path" type="string" required>
  Directory path to open
</ParamField>

<ParamField path="file_name" type="string">
  Optional filename to highlight/select
</ParamField>

<ResponseField name="result" type="void">
  Opens folder in Finder/Explorer/file manager
</ResponseField>

### Platform Behavior

- **Windows**: Uses `explorer.exe /select,<file>` to highlight file
- **macOS**: Uses `open -R <file>` to reveal in Finder
- **Linux**: Opens parent directory with `xdg-open` (no file selection)

### Usage Example

```typescript
// Open folder
await invoke('open_folder', {
  path: '/Users/me/Downloads'
})

// Open folder and highlight file
await invoke('open_folder', {
  path: '/Users/me/Downloads',
  file_name: 'video.mp4'
})
```

## get_default_download_path

Get the system's default download folder.

```typescript
const defaultPath = await invoke<string>('get_default_download_path')
```

<ResponseField name="defaultPath" type="string">
  Path to user's Downloads folder
</ResponseField>

### Usage Example

```typescript

console.log(`Default downloads: ${path}`)
// Windows: C:\Users\Username\Downloads
// macOS: /Users/username/Downloads
// Linux: /home/username/Downloads
```

## get_download_folder_size

Calculate total size of a download folder.

```typescript
const sizeBytes = await invoke<number>('get_download_folder_size', {
  path: string
})
```

<ParamField path="path" type="string" required>
  Folder path to analyze
</ParamField>

<ResponseField name="sizeBytes" type="number">
  Total size in bytes
</ResponseField>

### Usage Example

```typescript
const bytes = await invoke<number>('get_download_folder_size', {
  path: '/Users/me/Downloads'
})

const mb = (bytes / 1024 / 1024).toFixed(2)
console.log(`Folder size: ${mb} MB`)
```

## transcode_for_playback

Transcode a media file to MP4 for web playback.

```typescript
const result = await invoke<TranscodeResult>('transcode_for_playback', {
  input_path: string,
  force?: boolean
})
```

<ParamField path="input_path" type="string" required>
  Path to media file
</ParamField>

<ParamField path="force" type="boolean">
  Force transcode even if format is supported (default: false)
</ParamField>

<ResponseField name="TranscodeResult" type="object">
  <Expandable title="properties">
    <ResponseField name="output_path" type="string">
      Path to transcoded file (or original if not transcoded)
    </ResponseField>
    
    <ResponseField name="was_transcoded" type="boolean">
      Whether file was actually transcoded
    </ResponseField>
  </Expandable>
</ResponseField>

### Usage Example

```typescript
const result = await invoke<TranscodeResult>('transcode_for_playback', {
  input_path: '/Users/me/video.mkv',
  force: false
})

if (result.was_transcoded) {
  console.log(`Transcoded to: ${result.output_path}`)
} else {
  console.log('Format already supported, no transcode needed')
}
```
